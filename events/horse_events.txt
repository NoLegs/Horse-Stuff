namespace = horse

# Gives ruler the immortal trait.
country_event = {
	id = horse.1
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		has_civic = civic_alicorn_leader
	}

	immediate = {
		ruler = { add_trait = leader_trait_alicorn_ruler }
	}
}
