civic_alicorn_leader = {
	modification = no
	potential = {
		ethics = { 
			NOT = { value = ethic_gestalt_consciousness } 
		}
		authority = { 
			value = auth_imperial
			NOT = { value = auth_corporate } 
		}
	}
	
	possible = {
		civics = {
			NOT = { 
				value =	civic_solar_goddess
				value = civic_dark_imperator 
				value = civic_changeling_hive
				value = civic_fanatic_purifiers				
			}
		}
	}
	random_weight = { 
		base = 0 
		modifier = {
			factor = 0
			is_species_class = HRS
		}
	}
	modifier = {
		pop_government_ethic_attraction = 0.50
		pop_citizen_happiness = 0.15
	}
}

civic_changeling_hive = {
	modification = no
	potential = {
		ethics = { 
			NOT = { value = ethic_gestalt_consciousness } 
		}
		authority = { 
			OR = {
				value = auth_imperial
				value = auth_dictatorial
			}
			NOT = { value = auth_corporate } 
		}
	}
	possible = {
		civics = {
			NOT = { 
				value =	civic_solar_goddess
				value = civic_dark_imperator 
				value = civic_alicorn_leader 
				value = civic_fanatic_purifiers
			}
		}
	}
	random_weight = { 
		base = 0 
		modifier = {
			factor = 0
			is_species_class = HRS
		}
	}
	modifier = {
		country_base_influence_produces_add = 0.5
		leaders_upkeep_mult = -0.33
		planet_jobs_worker_produces_mult = 0.05
	}
}
